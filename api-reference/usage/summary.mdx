# Get Usage Summary

Returns aggregated usage summary including totals and per-model breakdown.

## Request

<ParamField query="workOrderId" type="string">
  Filter by work order ID
</ParamField>

<ParamField query="runId" type="string">
  Filter by run ID
</ParamField>

<ParamField query="model" type="string">
  Filter by model name
</ParamField>

<ParamField query="startDate" type="string">
  Start date filter (ISO 8601)
</ParamField>

<ParamField query="endDate" type="string">
  End date filter (ISO 8601)
</ParamField>

```bash
curl "https://agentgate.mynewapi.com/api/v1/usage/summary?startDate=2024-01-01&endDate=2024-01-31" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

## Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object" required>
  Usage summary data

  <Expandable title="properties">
    <ResponseField name="totalInputTokens" type="number" required>
      Total input tokens used
    </ResponseField>

    <ResponseField name="totalOutputTokens" type="number" required>
      Total output tokens used
    </ResponseField>

    <ResponseField name="totalCachedInputTokens" type="number" required>
      Total cached input tokens
    </ResponseField>

    <ResponseField name="totalCostUsd" type="number" required>
      Total estimated cost in USD
    </ResponseField>

    <ResponseField name="executionCount" type="number" required>
      Total number of executions
    </ResponseField>

    <ResponseField name="workOrderCount" type="number" required>
      Number of unique work orders
    </ResponseField>

    <ResponseField name="totalDurationMs" type="number" required>
      Total execution duration in milliseconds
    </ResponseField>

    <ResponseField name="byModel" type="object" required>
      Usage breakdown by model
    </ResponseField>

    <ResponseField name="periodStart" type="string" required>
      Period start date (ISO 8601)
    </ResponseField>

    <ResponseField name="periodEnd" type="string" required>
      Period end date (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>
